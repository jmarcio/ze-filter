#! /bin/sh


RSERVER="foss.jose-marcio.org:1873"

MAKE=make
RSYNC=rsync

CDBDIR=/var/ze-filter/cdb
BINDIR=/usr/bin

cd $CDBDIR

FILES2GET="ze-toolbox/ze-bayes-md5.txt"
for file in $FILES2GET
do
  $RSYNC -avz rsync://$RSERVER/$file .
  [ "$?" = "0" ] && $MAKE -f Makefile
done

LR2GET="lr-data/"
$RSYNC -avz rsync://$RSERVER/$LR2GET $LR2GET
[ "$?" = "0" ] && ze-ndc reload lrdata
